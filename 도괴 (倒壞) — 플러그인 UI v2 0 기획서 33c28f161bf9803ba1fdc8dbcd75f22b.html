1. 개요

1.1 목표

기존 v1.0 스마트폰 UI를 확장하여 스마트폰 팝업 + 방 탐색 핫스팟을 통합 구현합니다.

1.2 핵심 원칙

원칙설명
상태 유지팝업 열림/닫힘, 탭 선택 상태를 플러그인이 관리
실시간 동기화Lua chatVar를 JS가 직접 읽어 UI에 반영
토큰 최소화AI는{PHONE},{HOTSPOT} 태그만 출력, 렌더링은 플러그인

// chatVar 동기화
async function syncChatVars() {
const vars = [
'cv_floor', 'cv_step', 'cv_battery', 'cv_signal',
'cv_companion', 'cv_aff_seo', 'cv_aff_oh', 'cv_aff_kang',
'cv_aff_lee', 'cv_aff_jung', 'cv_hotspot_done', 'cv_phone_open'
];

for (const v of vars) {
const el = document.querySelector(`[data-var="${v}"]`);
if (el) {
const value = await getChatVar(v);
el.textContent = formatValue(v, value);
}
}
}

// 핫스팟 렌더링
function renderHotspot(floor, step) {
const container = document.querySelector('.hotspot-container');
if (!container) return;

const objects = FLOOR_OBJECTS[floor];
if (!objects) return;

// step 조건 필터링
const available = objects.filter(obj => {
if (obj.step_min && step < obj.step_min) return false;
if (obj.step_max && step > obj.step_max) return false;
return true;
});

// 버튼 생성
container.innerHTML = available.map(obj => {
const done = isDone(obj.id);
const hasReq = checkRequirement(obj.requires);

return `
<button class="hotspot-btn ${done ? 'hotspot-btn--done' : ''} ${obj.danger ? 'hotspot-btn--danger' : ''}"
data-action="${obj.type}"
data-id="${obj.id}"
${!hasReq ? 'disabled' : ''}>
${obj.label}
</button>
`;
}).join('');

// 이벤트 바인딩
container.querySelectorAll('.hotspot-btn').forEach(btn => {
btn.addEventListener('click', () => handleHotspotClick(btn));
});
}

// 버튼 클릭 → Lua 전달
async function handleHotspotClick(btn) {
const action = btn.dataset.action;
const id = btn.dataset.id;

// RisuAI API로 Lua 트리거 호출
await triggerLua(`onButtonClick('${action}', '${id}')`);
}

// chatVar 동기화
async function syncChatVars() {
const vars = [
'cv_floor', 'cv_step', 'cv_battery', 'cv_signal',
'cv_companion', 'cv_aff_seo', 'cv_aff_oh', 'cv_aff_kang',
'cv_aff_lee', 'cv_aff_jung', 'cv_hotspot_done', 'cv_phone_open'
];

for (const v of vars) {
const el = document.querySelector(`[data-var="${v}"]`);
if (el) {
const value = await getChatVar(v);
el.textContent = formatValue(v, value);
}
}
}

// 핫스팟 렌더링
function renderHotspot(floor, step) {
const container = document.querySelector('.hotspot-container');
if (!container) return;

const objects = FLOOR_OBJECTS[floor];
if (!objects) return;

// step 조건 필터링
const available = objects.filter(obj => {
if (obj.step_min && step < obj.step_min) return false;
if (obj.step_max && step > obj.step_max) return false;
return true;
});

// 버튼 생성
container.innerHTML = available.map(obj => {
const done = isDone(obj.id);
const hasReq = checkRequirement(obj.requires);

return `
<button class="hotspot-btn ${done ? 'hotspot-btn--done' : ''} ${obj.danger ? 'hotspot-btn--danger' : ''}"
data-action="${obj.type}"
data-id="${obj.id}"
${!hasReq ? 'disabled' : ''}>
${obj.label}
</button>
`;
}).join('');

// 이벤트 바인딩
container.querySelectorAll('.hotspot-btn').forEach(btn => {
btn.addEventListener('click', () => handleHotspotClick(btn));
});
}

// 버튼 클릭 → Lua 전달
async function handleHotspotClick(btn) {
const action = btn.dataset.action;
const id = btn.dataset.id;

// RisuAI API로 Lua 트리거 호출
await triggerLua(`onButtonClick('${action}', '${id}')`);
}
